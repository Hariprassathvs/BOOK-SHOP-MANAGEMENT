    public void Load()
    {
        try {
            pat=con.prepareStatement("select *from stu3");
            ResultSet rs=pat.executeQuery();
            ResultSetMetaData rss=rs.getMetaData();
            int c;
            c=rss.getColumnCount();
            ds=(DefaultTableModel)jTable1.getModel();
            ds.setRowCount(0);
            while(rs.next())
            {
                Vector v=new Vector();
                for(int a=1;a<=c;a++)
                {
                    v.add(rs.getString("id"));
                    v.add(rs.getString("name"));
                    v.add(rs.getString("age"));
                    v.add(rs.getString("city"));
                }
                ds.addRow(v);
            }
        } catch (SQLException ex) {
            Logger.getLogger(bookshop.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

         try {

            pat=con.prepareStatement("update stu3 set name=?,age=?,city=? where id=?");
